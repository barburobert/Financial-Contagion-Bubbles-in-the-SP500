# 5. Defining the ADF right-tail test statistic

def adf_right_tail(series):
    result = adfuller(
        series,
        maxlag=0,
        regression="c",
        autolag=None
    )
    return result[0]
#This function computes the Augmented Dickey-Fuller test statistic for a given time series using a fixed specification with no lag augmentation and an intercept term.
# It is used as the core building block for right-tailed unit root testing, where unusually large test statistics may indicate explosive behavior in asset prices.

# 6. Defining the GSADF procedure

def gsadf(series, min_window):
    T = len(series)
    stats = []

    for r2 in range(min_window, T):
        adf_vals = []

        for r1 in range(0, r2 - min_window + 1):
            sub_series = series[r1:r2]
            adf_vals.append(adf_right_tail(sub_series))

        stats.append(np.max(adf_vals))

    return np.array(stats)
#This function implements the Generalized Sup Augmented Dickey-Fuller (GSADF) procedure.
#For each possible endpoint of the sample, it computes the ADF statistic over multiple rolling and expanding subsamples, then retains the maximum value.
#This allows the test to detect episodes of mildly explosive behavior that may emerge at different points in time.

# 9. Defining the SADF procedure

def sadf(series, min_window):
    T = len(series)
    adf_vals = []

    for r2 in range(min_window, T):
        sub_series = series[:r2]
        adf_vals.append(adf_right_tail(sub_series))

    return np.max(adf_vals)
#This function computes the Sup Augmented Dickey-Fuller (SADF) statistic by applying the ADF test recursively from the beginning of the sample to progressively larger subsamples.
#The maximum test statistic is then retained, allowing the detection of explosive behavior that develops over time from a fixed starting point.

# 10. Bootstrapping critical values for the SADF test

def bootstrap_sadf(series, min_window, B=300):
    T = len(series)
    boot_stats = []

    for b in range(B):
        eps = np.random.normal(0, 1, T)
        y = np.cumsum(eps)  # random walk under H0
        boot_stats.append(sadf(y, min_window))

    return np.percentile(boot_stats, [90, 95, 99])

#This function generates bootstrap critical values for the SADF statistic under the null hypothesis of a random walk.
#For each simulation, a synthetic series is created as a cumulative sum of random shocks, the SADF statistic is computed, and the empirical distribution of these statistics is used to obtain the 90%, 95%, and 99% critical values.
